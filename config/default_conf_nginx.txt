server {
    listen 80 default_server;
    listen [::]:80 default_server;

    # 1. Donde está tu HTML y CSS
    root /var/www/html;
    index index.html;

    server_name _;

    location / {
        try_files $uri $uri/ =404;
    }

    # 2. Conexión con tu app.py (Flask)
    location /secretario/guardar {
        proxy_pass http://127.x.x.xxxxx;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}
